cmake_minimum_required(VERSION 3.19)
project(QtWidgetEsim1 LANGUAGES CXX)

find_package(Qt6 6.5 REQUIRED COMPONENTS Core Widgets)

qt_standard_project_setup()

qt_add_executable(QtWidgetEsim1
    WIN32 MACOSX_BUNDLE
    main.cpp
    mainwindow.cpp
    mainwindow.h
    mainwindow.ui
    page2.h page2.cpp page2.ui
    style.qss
    resource.qrc
)

qt_add_resources(RESOURCES_ADDED resource.qrc)
target_sources(QtWidgetEsim1 PRIVATE ${RESOURCES_ADDED})

target_link_libraries(QtWidgetEsim1
    PRIVATE
        Qt::Core
        Qt::Widgets
)

include(GNUInstallDirs)

install(TARGETS QtWidgetEsim1
    BUNDLE  DESTINATION .
    RUNTIME DESTINATION ${CMAKE_INSTALL_BINDIR}
    LIBRARY DESTINATION ${CMAKE_INSTALL_LIBDIR}
)

qt_generate_deploy_app_script(
    TARGET QtWidgetEsim1
    OUTPUT_SCRIPT deploy_script
    NO_UNSUPPORTED_PLATFORM_ERROR
)
install(SCRIPT ${deploy_script})
